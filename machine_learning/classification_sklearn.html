The below function, k_nearest_neighbor takes 6 paramters¶

train: document term matrix in a pandas dataframe
test: a document term matrix in a pandas dataframe to be classified 
train_labels: a dataframe containing the class labels of the training data
test_labels: a dataframe containing the class labels of the test documents
measure: either 'euclidean' or 'cosine' for the desired similarity measure
k: the number of Ks to be used

The below function combines components of parts a-b. The below function will take a dataframe of test vectors and compare them against the training data, returning the number of correct predictions, observations and an overall accuracy measure. If desired, a single array could be input as well. Please note, the function automatically converts each row of a dataframe to a numpy array before comparison. The function can be altered to perform Euclidean distance or cosine similairty and the desired number of Ks can also be chosen. When sorting, if 2 neighbors have the same distance one will be randomly selected for the nearest neighbor. For instance, if neighbors 5 and 6 were the same distance and the k was set to 5, the function would choose one of the tying observations at random. Another option could have been to select them both but given the accuracies were so high, I was satisfied with the initial approach.

def k_nearest_neighbor(train, test, train_labels, test_labels, measure, k):

    #initialize a dataframe to store the similarity measure for each training document
    global final
    ratings = {}
    final = pandas.DataFrame(columns = ('Actual_Class','Predicted_Class'))

    #iterate through the training and test dataframes
    for index_test,row_test in test.iterrows():
        for index_train,row_train in train.iterrows():

            #each loop, we reinitialize the similarity measures
            norm_test = 0
            norm_train = 0
            dot = 0
            euclidean = 0
            cosine = 0

            #convert each row to a numpy array so we can use the linalg and dot functions
            row_test = numpy.array(row_test)
            row_train = numpy.array(row_train)

            #get the euclidean distance
            dist_euclidean = numpy.sqrt(((row_train - row_test)**2).sum(axis=0))
            
            #get the norm for the test and training documents
            norm_test = numpy.linalg.norm(row_test)
            norm_train = numpy.linalg.norm(row_train)

            #get the dot product
            dot = numpy.dot(row_test,row_train)

            #get the cosine similarity
            cosine = dot/(norm_test * norm_train)
            dist_cosine = 1 - cosine
            
            #if cosine similarity, append to the list
            if measure == 'cosine':
                if index_test not in ratings:
                    ratings[index_test] = []
                    ratings[index_test].append([index_train,dist_cosine])
                else:
                    ratings[index_test].append([index_train,dist_cosine])
                    
            #otherwise use euclidean distance      
            else:
                if index_test not in ratings:
                    ratings[index_test] = []
                    ratings[index_test].append([index_train,dist_euclidean])
                else:
                    ratings[index_test].append([index_train,dist_euclidean])
    
    #begin looping thorugh the ratings dictionary
    for key,value in ratings.items():
        lst = []
        rating = []
        knn = []
        final_ratings = []
        
        #for each value in the list, we sort based on the similarity measure in decending order
        for item in value:
            lst.append(item)
            lst = sorted(lst,key = lambda tup: tup[1])
        
        #we chop the top k measuers off
        final_list = lst[:k]
        
        #this gives a list of the nearest neighbors
        for item1 in final_list:
            knn.append(item1[0])

        #this gives the final labels
        for item1 in knn:
            final_ratings.append(train_labels.iloc[item1][1])
            
        #we use the mode to get the predicted class
        predicted = max(set(final_ratings), key=final_ratings.count)

        #we append the actual rating and predicted to the final dataframe
        actual_rating = test_labels.iloc[key][1]
        final.loc[key] = (actual_rating, predicted) 
        
        #print("Observation: ", key)
        #print("Actual Rating: ", actual_rating)
        #print("Predicted Rating: ",predicted)
        #print('\n')
        #print(key,final_list)
        #print(actual_rating,predicted)
        
    final['boolean'] = final['Actual_Class'] == final['Predicted_Class'] 
    correct_class = float(final[final.boolean == True].count()["boolean"])
    observations = float(len(final.index))
    accuracy = correct_class/observations
    
    print('K: ', k)
    print('Similarity Measure: ', measure)
    print('\n')
    print("Correct: ",correct_class)
    print("Observations: ", observations)
    print("Accuracy: ", accuracy)

def roc(train, test, train_labels, test_labels, measure):
    
    #here we create prototype vectors for rocchios method
    tester = pandas.concat([train,train_labels], axis=1)
    prototypes =  pandas.DataFrame(tester.groupby(["class"]).mean())

    #initialize a dataframe to store the similarity measure for each training document
    global final
    ratings = {}
    final = pandas.DataFrame(columns = ('Actual_Class','Predicted_Class'))

    #iterate through the training and test dataframes
    for index_test,row_test in test.iterrows():
        for index_p,row_p in prototypes.iterrows():

            #each loop, we reinitialize the similarity measures
            norm_test = 0
            norm_p = 0
            dot = 0
            euclidean = 0
            cosine = 0

            #convert each row to a numpy array so we can use the linalg and dot functions
            row_test = numpy.array(row_test)
            row_p = numpy.array(row_p)

            #get the euclidean distance
            euclidean = numpy.sqrt(((row_p - row_test)**2).sum(axis=0))
            dist_euclidean = euclidean

            #get the norm for the test and training documents
            norm_test = numpy.linalg.norm(row_test)
            norm_p = numpy.linalg.norm(row_p)

            #get the dot product
            dot = numpy.dot(row_test,row_p)

            #get the cosine similarity
            cosine = dot/(norm_test * norm_p)
            dist_cosine = 1 - cosine
            
            #if cosine similarity, append to the list
            if measure == 'cosine':
                if index_test not in ratings:
                    ratings[index_test] = []
                    ratings[index_test].append([index_p,dist_cosine])
                else:
                    ratings[index_test].append([index_p,dist_cosine])
                    
            #otherwise use euclidean distance      
            else:
                if index_test not in ratings:
                    ratings[index_test] = []
                    ratings[index_test].append([index_p,dist_euclidean])
                else:
                    ratings[index_test].append([index_p,dist_euclidean])
            
    #begin looping thorugh the ratings dictionary
    for key,value in ratings.items():
        lst = []
        rating = []
        knn = []
        final_ratings = []
        
        #for each value in the list, we sort based on the similarity measure in decending order
        for item in value:
            lst.append(item)
            lst = sorted(lst,key = lambda tup: tup[1])
        
        #we chop the top k measuers off
        final_list = lst[:1]
        
        for item1 in final_list:
            knn.append(item1[0])
            
        #this gives the final labels
        for item1 in knn:
            final_ratings.append(test_labels.iloc[item1][0])
     
        #we use the mode to get the predicted class
        predicted = lst[0][0]

        #we append the actual rating and predicted to the final dataframe
        actual_rating = test_labels.iloc[key][0]
        final.loc[key] = (actual_rating, predicted) 
        
        print("Observation: ", key )
        print("Actual Rating: ", actual_rating)
        print("Predicted Rating: ", predicted)
        print("\n")
        
    final['Predicted_Class'] = final['Actual_Class'] == final['Predicted_Class'] 
    correct_class = float(final[final.Predicted_Class == True].count()["Predicted_Class"])
    observations = float(len(final.index))
    accuracy = correct_class/observations
    
    print('\n')
    print("Correct: ",correct_class)
    print("Observations: ", observations)
    print("Accuracy: ", accuracy)
    print('\n')
